    \textbf{Question 1-(xi)[ICSE 2017]:}
    \begin{problem}
    if a+ib = $\frac{x+iy}{x-iy}$ \\
    prove that  $a^2 + b^2$ = 1 and $\frac{b}{a} = \frac{2xy}{x^2 - y^2}$
    \end{problem}
    
\newcommand{\solution}{\noindent \textbf{Solution: }}
        \begin{align}
        a+ib = \frac{x+iy}{x-iy}
        \end{align}
        
        let r be the magnitude of the given complex number a+ib
        
       so \begin{align}
       r= \sqrt{ a^2+b^2}
       \end{align}
       
        \begin{align}  
        &\tan{\theta}=\frac{b}{a}\\     
        &x+iy = s(\cos{\alpha}+i\sin{\alpha})\\   
        &\tan{\alpha} = \frac{y}{x}
        \end{align}
    
    \begin{align}
  &r(\cos{\theta}+i\sin{\theta})=\dfrac{\cos{\alpha}+i\sin{\alpha}}{\cos{\alpha}-i\sin{\alpha}} 
   \end{align}
   
    multiplying denominator on both numerator and denominator we get 
    
      \begin{align}
      \cos^2{\alpha}-\sin^2{\alpha}+2i\sin{\alpha}\times\cos{\alpha}
      \end{align}on RHS
       
       \begin{align}
        r(\cos{\theta+i\sin{\theta}})=\cos{2\alpha}+i\sin{2\alpha}\\
       \implies r=1 
       \end{align}
     
       magnitude of complex number on RHS is 1 so definitely magnitude on LHS is 1
      
       so comparing both LHS and RHS  
       \begin{align}
       \theta=2\alpha 
       \end{align}
       
       applying $\tan$ on both sides, we get 
       \begin{align}
       \tan{\theta}=\tan{2\alpha}
       \end{align}
       
     \begin{align}
       &\frac{b}{a} = \frac{2\tan{\alpha}}{1-\tan^2{\alpha}}\\
       &\frac{b}{a} = \frac{2y/x}{1-y^2/x^2}\\
       &\implies \frac{b}{a} = \frac{2xy}{x^2-y^2}
     \end{align}
